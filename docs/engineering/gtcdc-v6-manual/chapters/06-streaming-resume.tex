\chapter{流式 Process2FGearChunkCut 与双 hash Resume}
\label{ch:streaming-resume}

\section{状态机字段}

\texttt{GtCdcStreamState} 中 v6 专用字段：
\begin{itemize}[nosep]
  \item \texttt{tf\_gear\_h}, \texttt{tf\_norm\_h}：resume 注入值；
  \item \texttt{tf\_h\_valid}：是否跳过 \texttt{InitWindowHash}；
  \item \texttt{tf\_scan\_abs}：绝对字节坐标下回退扫描起点（跨 512\,KB feed）。
\end{itemize}

\texttt{TwoFClearResume} 在切点确认或 max-cut 后清空上述状态。

\section{Process2FGearChunkCut 逻辑}

\begin{lstlisting}[caption={Process2FGearChunkCut（\filepath{gt\_cdc\_streaming.cc} 结构）},
  label={lst:process2f}]
bool Process2FGearChunkCut(..., size_t pos_bias, uint64_t* probes, ...) {
  // 1. tail <= min_size：末段合并或 stop_outer
  // 2. scan_start = pos + min_size；若 tf_scan_abs 有效则前移
  Build2FMasks(cfg.avg_size, cfg.nc_level, &mask_lo, &mask_hi, &norm_shift);
  GtCdcScanGear2FView(view, scan_start, cut, w, mask_lo, mask_hi, norm_shift,
      cfg.beta_table, cfg.norm_table, &cut, &found,
      state->tf_gear_h, state->tf_norm_h, state->tf_h_valid,
      &h_gear_out, &h_norm_out, probes);
  if (!found && remaining > max_size) { cut = pos + max_size; TwoFClearResume; }
  else if (!found && !is_last) {
    state->tf_gear_h = h_gear_out; state->tf_norm_h = h_norm_out;
    state->tf_h_valid = true;
    state->tf_scan_abs = pos_bias + scan_end;
    *stop_outer = true;  // 等待下一段 feed
  } else if (found) { TwoFClearResume; }
  *cut_out = cut;
}
\end{lstlisting}
