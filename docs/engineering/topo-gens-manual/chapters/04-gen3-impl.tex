\section{PrimaryTopoDelta：数学 $=$ 代码}

\begin{lstlisting}[caption={三边差分（\filepath{topo\_hom\_scan.cc}）},label={lst:primary-topo-delta}]
inline int32_t PrimaryTopoDelta(const uint8_t* keys, uint32_t w,
                                uint8_t key_in, uint32_t ed_old) {
  uint32_t ed_new = ed_old;
  if (keys[0] != keys[1]) --ed_new;
  if (w == 2) {
    if (keys[1] != key_in) ++ed_new;
  } else {
    const uint8_t old_tail = keys[w - 1];
    if (keys[w - 2] != old_tail) --ed_new;
    if (keys[w - 2] != key_in) ++ed_new;
    if (key_in != keys[0]) ++ed_new;
  }
  return static_cast<int32_t>(ed_new) - static_cast<int32_t>(ed_old);
}
\end{lstlisting}

\section{ProbeHomAt 热路径}

\begin{lstlisting}[caption={仅 primary 时 $O(w)$ 键化},label={lst:probe-hom}]
inline bool ProbeHomAt(uint32_t* h, uint32_t mask, uint32_t w,
                       const uint32_t gear[256], const uint8_t* data,
                       size_t pos, size_t* out_cut) {
  const uint32_t g = gear[data[pos]];
  if ((*h & mask) == 0) {
    uint8_t keys[kTopoMaxWindow]{};
    uint32_t ed = 0;
    uint8_t prev = TopoKey(gear, data[pos - w]);
    keys[0] = prev;
    for (uint32_t i = 1; i < w; ++i) {
      const uint8_t k = TopoKey(gear, data[pos - w + i]);
      keys[i] = k;
      ed += prev != k;
      prev = k;
    }
    if (PrimaryTopoDelta(keys, w, static_cast<uint8_t>(g & 0xFFu), ed) != 0) {
      *out_cut = pos;
      return true;
    }
  }
  *h = (*h << 1) + g - gear[data[pos - w]];
  return false;
}
\end{lstlisting}

\texttt{RunHomScanLoop} 在 AVX2 下按 8 字节批调用 \texttt{ProbeHomAt}，切点集合与标量路径相同。

\section{mask 构造印证}

\begin{lstlisting}[caption={BuildTopoMasks（标定优先）},label={lst:build-topo-masks}]
void BuildTopoMasks(uint32_t avg_size, uint16_t calib_permille,
                    uint8_t topo_shift, uint32_t* mask_out) {
  if (calib_permille > 0) {
    const double p_topo = max(calib_permille / 1000.0, 0.001);
    const uint32_t effective =
        max<uint32_t>(uint32_t(double(avg_size) * p_topo), 1u);
    *mask_out = gtcdc_internal::BuildMask(effective);
    return;
  }
  const uint32_t shifted = max<uint32_t>(avg_size >> topo_shift, 1u);
  *mask_out = gtcdc_internal::BuildMask(shifted);
}
\end{lstlisting}

\texttt{CalibrateTopoPermille}：先用 \texttt{topo\_shift} 临时 mask 扫描半样本，统计
\texttt{topo\_hits/probes}，$\mathrm{permille}=\mathrm{clamp}(1000\cdot\hat P,\,1,\,1000)$。

\section{滚动 hash 初始化}

\begin{lstlisting}[caption={InitWindowHash},label={lst:init-window-hash}]
uint32_t InitWindowHash(const uint8_t* data, size_t end, uint32_t w,
                        const uint32_t gear[256]) {
  uint32_t h = 0;
  for (size_t i = end - w; i < end; ++i)
    h = (h << 1) + gear[data[i]];
  return h;
}
\end{lstlisting}

与 FastCDC / G-TCDC 同形，保证 Gear 表语义可对照；但 \texttt{table\_seed} 隔离，表内容不同。
