\subsection{样本统计公式（可复制验证）}
\begin{align}
  \bar{T} &= \frac{1}{N}\sum_{i=1}^{N} T_i, \\
  s &= \sqrt{\frac{1}{N-1}\sum_{i=1}^{N}(T_i-\bar{T})^2}, \\
  \mathrm{CV} &= 100\,\frac{s}{\bar{T}}, \\
  v_i &= \frac{B_{\mathrm{MB}}}{T_i}.
\end{align}

\subsection{95\% 置信区间（$N=10$，E: 组）}
\[
  \bar{T} \pm t_{0.975,\,9}\,\frac{s}{\sqrt{N}}
  \;=\; 19.025 \pm 2.262\times\frac{0.236}{\sqrt{10}}
  \;=\; 19.025 \pm 0.169\ \text{s}.
\]
